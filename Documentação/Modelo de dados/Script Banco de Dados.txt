CREATE TABLE Endereco (
id_endereco INTEGER GENERATED BY DEFAULT AS IDENTITY (START WITH 1, INCREMENT BY 1) PRIMARY KEY,
numero VARCHAR (10),
rua VARCHAR (100),
complemento VARCHAR (100),
cidade VARCHAR (100),
estado VARCHAR (100),
cep VARCHAR (100),
bairro VARCHAR (100),
FOREIGN KEY(id_endereco) REFERENCES Endereco(id_endereco)
);


CREATE TABLE Cliente (
id_cliente INTEGER GENERATED BY DEFAULT AS IDENTITY (START WITH 1, INCREMENT BY 1) PRIMARY KEY,
id_endereco INTEGER,
nome VARCHAR (100),
FOREIGN KEY(id_endereco) REFERENCES Endereco(id_endereco)
);


CREATE TABLE Forma_Pagamento (
id_forma_pagamento INTEGER GENERATED BY DEFAULT AS IDENTITY (START WITH 1, INCREMENT BY 1) PRIMARY KEY,
descricao VARCHAR (100)
);


CREATE TABLE Forma_Envio (
id_forma_envio INTEGER GENERATED BY DEFAULT AS IDENTITY (START WITH 1, INCREMENT BY 1) PRIMARY KEY,
descricao VARCHAR (100)
);


CREATE TABLE Produto (
id_produto INTEGER GENERATED BY DEFAULT AS IDENTITY (START WITH 1, INCREMENT BY 1) PRIMARY KEY,
nome VARCHAR (100),
quantidade INTEGER,
valor NUMERIC,
descricao VARCHAR (100)
);


CREATE TABLE Encomenda (
id_encomenda INTEGER GENERATED BY DEFAULT AS IDENTITY (START WITH 1, INCREMENT BY 1) PRIMARY KEY,
id_produto INTEGER,
id_forma_pagamento INTEGER,
id_forma_envio INTEGER,
ordem_servico INTEGER,
FOREIGN KEY(id_produto) REFERENCES Produto(id_produto),
FOREIGN KEY(id_forma_pagamento) REFERENCES Forma_Pagamento(id_forma_pagamento),
FOREIGN KEY(id_forma_envio) REFERENCES Forma_Envio(id_forma_envio)
);
