 --- criar tabela com primary key auto incremento começando de um numero qualquer e com um incremento qualquer ---
CREATE TABLE users (
  id INTEGER GENERATED BY DEFAULT AS IDENTITY(START WITH 100, INCREMENT BY 1) PRIMARY KEY,
  name VARCHAR(30),
  email  VARCHAR(50)
);

 --- criar tabela com mais de uma foreign key ---

CREATE TABLE encomenda (
id_encomenda INTEGER IDENTITY PRIMARY KEY,
id_forma_pagamento INTEGER,
id_forma_envio INTEGER,
FOREIGN KEY(id_forma_pagamento) REFERENCES forma_pagamento(id_forma_pagamento),
FOREIGN KEY(id_forma_envio) REFERENCES forma_envio(id_forma_envio)
);
